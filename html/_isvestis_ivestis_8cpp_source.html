24// Skaičiuoja UTF-8 simbolių skaičių (display width), ne baitų
25size_t utf8_display_width(const string& str) {
26 size_t width = 0;
27 for (size_t i = 0; i < str.size(); ) {
28 unsigned char c = str[i];
29 if ((c & 0x80) == 0) {
30 // ASCII simbolis (1 baitas)
31 i += 1;
32 } else if ((c & 0xE0) == 0xC0) {
33 // 2 baitų UTF-8 simbolis
34 i += 2;
35 } else if ((c & 0xF0) == 0xE0) {
36 // 3 baitų UTF-8 simbolis
37 i += 3;
38 } else if ((c & 0xF8) == 0xF0) {
39 // 4 baitų UTF-8 simbolis
40 i += 4;
41 } else {
42 i += 1; // Netikėtas baitas
43 }
44 width++;
45 }
46 return width;
47}

49// Papildo eilutę tarpais iki nurodyto pločio (pagal display width)
50string pad_utf8(const string& str, size_t width) {
51 size_t display_width = utf8_display_width(str);
52 if (display_width >= width) {
53 return str;
54 }
55 return str + string(width - display_width, ' ');
56}
